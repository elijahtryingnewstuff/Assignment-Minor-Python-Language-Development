// 1. Basic local scope  
function test() {  
  local = 42  
  print local  
}  
test()  

// 2. Shadowing globals  
global = 100  
function shadow() {  
  global = 200  
  print global  
}  
shadow()  
print global  

// 3. Locals in loops  
function counter() {  
  i = 0  
  while (i < 3) {  
    local = i * 2  
    print local  
    i = i + 1  
  }  
}  
counter()  

// 4. Locals in conditionals  
function check(x) {  
  if (x > 0) {  
    result = "Positive"  
  } else {  
    result = "Non-positive"  
  }  
  print result  
}  
check(5)
