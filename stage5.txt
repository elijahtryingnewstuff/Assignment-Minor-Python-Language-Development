# Test 9: Nested loops
outer = 1
while (outer <= 2) {
    inner = 1
    while (inner <= 2) {
        print "Outer:" + to_string(outer) + " Inner:" + to_string(inner)
        inner = inner + 1
    }
    outer = outer + 1
}

# Test 10: Complex control flow
running = true
attempts = 0
while (running) {
    cmd = input("Enter command (stop/continue): ")
    if (cmd == "stop") {
        running = false
    } else {
        attempts = attempts + 1
        if (attempts >= 3) {
            print "Max attempts reached"
            running = false
        } else {
            print "Try again (" + to_string(attempts) + "/3)"
        }
    }
}
