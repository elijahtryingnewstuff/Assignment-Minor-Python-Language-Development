# Create an empty dictionary
d = {}
print d == {}                    # Expected: true

# Assign different types of values
d["name"] = "Marina"
d["age"] = 20
d["is_student"] = true
d["grades"] = [10, 9, 8]
d["profile"] = {"city": "Athens", "code": 1337}

print d["name"]                 # Expected: "Marina"
print d["age"] + 2             # Expected: 22
print d["is_student"] == true  # Expected: true
print d["grades"][0]           # Expected: 10
print d["profile"]["city"]     # Expected: "Athens"

# Overwrite existing key
d["name"] = "Dr. Marina"
print d["name"]                # Expected: "Dr. Marina"

# Mixed key types (number, string, boolean)
d[1] = "one"
d[true] = "yes"
print d[1]                     # Expected: "one"
print d[true]                  # Expected: "yes"

# Key not found
print d["unknown"]            # Expected: Error: Key 'unknown' not found

# Invalid key type (e.g., list as key)
bad = {[1, 2]: "fail"}        # Expected: Error: Dictionary keys must be strings, numbers, or booleans

# Dictionary inside dictionary
deep = {"outer": {"inner": {"value": 123}}}
print deep["outer"]["inner"]["value"]  # Expected: 123

# Dictionary comparison
a = {"x": 1, "y": 2}
b = {"y": 2, "x": 1}
print a == b                  # Expected: true

# Dictionary addition (bonus if supported)
combined = a + {"z": 3}
print combined                # Expected: {"x": 1, "y": 2, "z": 3}
